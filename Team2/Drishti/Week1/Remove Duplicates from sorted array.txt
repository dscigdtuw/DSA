class Solution {
public:
    int removeDuplicates(vector<int>& a) {
        int i=0,j=0,n=a.size();
        if(n==0 || n==1)
            return n;
        
        while(j<n){
            while(a[i]==a[j] && j<n-1){
               j++; 
            }
            if(a[i]==a[j])
                return ++i;
            
            else
                a[++i]=a[j];
        }
            
        return i;
    }
};

Time Complexity: O(n)
Space Complexity: O(1)
