#include<iostream>
#include<vector>

using namespace std;

class Solution {
    
public:
    void rotate(vector<int>& nums, int k) 
    {
            k=k%nums.size();
            reverse(nums, 0, nums.size()-1);
            reverse(nums, 0, k-1);
            reverse(nums, k, nums.size()-1); 
    }
    
    void reverse(vector<int>& nums1, int i, int j)
    {
      
        while(i<j)
        {
            swap(nums1,i,j);
            i++;
            j--;
        }

    }
    
    void  swap(vector<int>& nums2,int x,int y)
    {
        int temp;
        temp=nums2[x];
        nums2[x]=nums2[y];
        nums2[y]=temp;
    }

}s;
