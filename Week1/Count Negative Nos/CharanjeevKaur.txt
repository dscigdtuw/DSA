class Solution {
public:
    int countNegatives(vector<vector<int>>& grid) 
    {
        int m  = grid.size();
        int n = 0, count = 0;
        
        for(int i = m-1;i >= 0;i--)
        {
            n = grid[i].size();
            for(int j = n-1;j >= 0;j--)
            
                if(grid[i][j] >= 0)
                    break;
                else
                    count++;
            
        }
           
        return count;
    }
};
